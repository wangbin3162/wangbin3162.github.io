# Object 对象方法

Object.assign 复制对象、合并对象

语法:Object.assign(target, ...sources)

  • 参数
    • target: 目标对象
    • sources: 源对象
  • 返回值
    • 目标对象
// 克隆对象
	var obj = {a: 1};
	var copy = Object.assign({}, obj);
	console.log(copy); // {a: 1};
// 合并对象
	var o1 = {a: 1};
	var o2 = {b: 2};
	var o3 = {c: 3};

	var obj = Object.assign(o1, o2, o3);
	console.log(obj); //{a: 1, b: 2, c: 3}
	console.log(o1); //{a: 1, b: 2, c: 3}, 目标对象被改变了

# js模块化

// AMD : require.js
// 全局define函数
// 全局require函数
// 依赖js会自动,异步加载
//a.js
define(function(){
    return {
        fn1:function(){
          console.log(fn1)
        }
    }
});
//b.js
define(['a.js'],function(a){
    return {
        fn2:function(){
            a.fn1();
            console.log(fn2);
        }
    }
});
//c.js
define(['b.js'],function(b){
    return {
        fn3:function(){
            b.fn2();
            console.log(fn3);
        }
    }
});
// CommonJS:
// nodejs模块化规范
// webpack ,npm安装依赖,库
// 不是异步加载,同步加载
//util.js
module.exports = {
    getFormateDate:function(date,type){
        return '2017-12-19';
    }
}
//a.js
var util = require('util.js');
module.exports = {
    formateDate:function(date){
        return util.getFormateDate(date,'xxxx-yy-mm');
    }
}

//main.js
var a = require('a.js');
a.formateDate(new Date());
